### 完整API测试脚本 (test_all_apis.py)
```python
"""RAG导购全量API自动化测试"""
import sys, io, requests, json
sys.stdout = io.TextIOWrapper(sys.stdout.buffer, encoding='utf-8')

BASE = "http://localhost:8000"
PASS, FAIL = 0, 0

def test(name, fn):
    global PASS, FAIL
    try:
        print(f"\n{'='*60}")
        print(f"[TEST] {name}")
        print('-'*60)
        fn()
        print(f"✅ PASS")
        PASS += 1
    except AssertionError as e:
        print(f"❌ FAIL: {e}")
        FAIL += 1
    except Exception as e:
        print(f"❌ ERROR: {type(e).__name__}: {e}")
        FAIL += 1

def assert_eq(a, b, msg=""):
    assert a == b, f"Expected {b}, got {a}. {msg}"

# --- 0. Health Check ---
test("Health Check", lambda: (
    lambda r: (assert_eq(r.status_code, 200, "health"), assert_eq(r.json()['status'], 'healthy'))
    )(requests.get(f"{BASE}/health", timeout=5))
))

# --- 1. Product Detail API ---
test("Product Detail: p_beauty_001", lambda: (
    lambda d: (
        assert_eq(d['product']['product_id'], 'p_beauty_001'),
        assert len(d['skus']) &gt;= 2
    )
    )(requests.get(f"{BASE}/api/v1/product/p_beauty_001", timeout=5).json())
))

# --- 2. 购物车 CRUD 全流程 ---
USER = f"test_user_{sys._getframe().f_lineno}"
def cart_crud_test():
    # 2.1 清空
    r = requests.delete(f"{BASE}/api/v1/cart?user_id={USER}", timeout=5)
    assert_eq(r.status_code, 200)
    
    # 2.2 Add item A
    r1 = requests.post(f"{BASE}/api/v1/cart", json={
        "user_id": USER, "product_id": "p_beauty_001", "sku_id": "s1",
        "title": "精华A", "brand": "BrandA", "unit_price": 500.0, "quantity": 1
    }, timeout=5)
    assert_eq(r1.status_code, 200)
    item1_id = r1.json()['item']['id']
    
    # 2.3 Add item B
    r2 = requests.post(f"{BASE}/api/v1/cart", json={
        "user_id": USER, "product_id": "p_beauty_002", "sku_id": "s2",
        "title": "精华B", "brand": "BrandB", "unit_price": 300.0, "quantity": 2
    }, timeout=5)
    assert_eq(r2.status_code, 200)
    
    # 2.4 Get cart → 1 + 2 = 3 件
    cart = requests.get(f"{BASE}/api/v1/cart?user_id={USER}", timeout=5).json()
    assert_eq(cart['total_count'], 3)
    assert_eq(cart['total_amount'], 500 + 300*2)
    
    # 2.5 Update qty → 把A从1改为3 → total=3+2=5
    requests.put(f"{BASE}/api/v1/cart/{item1_id}", json={"quantity":3}, timeout=5)
    cart2 = requests.get(f"{BASE}/api/v1/cart?user_id={USER}", timeout=5).json()
    assert_eq(cart2['total_count'], 5)
    
    # 2.6 Remove
    requests.delete(f"{BASE}/api/v1/cart/{item1_id}", timeout=5)
    cart3 = requests.get(f"{BASE}/api/v1/cart?user_id={USER}", timeout=5).json()
    assert_eq(cart3['total_count'], 2)
    
    # 2.7 Clear all
    requests.delete(f"{BASE}/api/v1/cart?user_id={USER}", timeout=5)
    cart4 = requests.get(f"{BASE}/api/v1/cart?user_id={USER}", timeout=5).json()
    assert_eq(cart4['total_count'], 0)

test("Cart CRUD Full Flow", cart_crud_test)

# --- 3. 对话式加购意图识别 ---
def chat_add_intent_test():
    SID = "test_session_add_intent"
    # 先推荐，让检索到一些商品
    r1 = requests.post(f"{BASE}/api/v1/chat", json={
        "session_id": SID, "query": "推荐精华"
    }, timeout=60)
    assert_eq(r1.status_code, 200)
    # 然后说"把第一个加到购物车"
    r2 = requests.post(f"{BASE}/api/v1/chat", json={
        "session_id": SID, "query": "把第一个加到购物车"
    }, timeout=60)
    assert_eq(r2.status_code, 200)
    data = r2.json()
    assert data.get('cart_action', {}).get('added_to_cart') == True, "cart_action.added_to_cart 必须为True"
    
    # 验证购物车中有商品
    cart = requests.get(f"{BASE}/api/v1/cart?user_id={SID}", timeout=5).json()
    assert cart['total_count'] &gt;= 1, "对话加购后购物车至少1件"

test("Chat Add To Cart Intent", chat_add_intent_test)

# --- 4. 反选排除测试 ---
def exclude_test():
    r = requests.post(f"{BASE}/api/v1/chat", json={
        "session_id": "exclude_test_1", "query": "推荐不要含酒精的精华"
    }, timeout=60)
    assert_eq(r.status_code, 200)
    data = r.json()
    assert '含酒精' in (data.get('exclude_keywords') or []), "exclude_keywords 必须包含'含酒精'"

test("Exclude / 不要语义识别", exclude_test)

# --- Summary ---
print("\n" + "="*60)
print(f"Total: {PASS + FAIL} | PASS: {PASS} | FAIL: {FAIL}")
print("="*60)
if FAIL &gt; 0:
    sys.exit(1)
```

运行：
```bash
cd d:\RAG导购
python test_all_apis.py
```

| 步骤 | 操作 | 预期结果 |

## 方案三：Android UI 专项测试清单

### 导航跳转树验证
```
ChatScreen
  ├─ 点商品卡片 → ProductDetailScreen
  │      ├─ 点← 返回 → ChatScreen
  │      └─ 点右上角购物车 → CartScreen
  │              └─ 点商品 → ProductDetailScreen
  └─ 点右上角购物车 → CartScreen
          ├─ 点← 返回 → ChatScreen
          └─ 点商品 → ProductDetailScreen
```
- 测试：在任意页面按系统 Back 键（模拟器右侧←），都能按导航栈层级正确退回，不会闪退白屏

### 多方向/不同分辨率适配测试
- 竖屏横屏旋转：旋转屏幕后，所有页面 UI 布局不溢出、不重叠，商品卡片横向滚动正常
- 大字体模式（系统设置 → 显示 → 字号调到最大）：文字不溢出卡片、按钮文字不被截断

### 边界场景异常处理
| 异常场景 | 操作 | 预期行为 |
|---------|------|---------|
| 网络断开 | 关闭模拟器 WIFI，点「加入购物车」 | 弹出 Snackbar：「加购失败: 连接失败」，App 不崩溃 |
| 购物车为空 | 清空所有项 → 进入购物车页 | 居中显示大 ShoppingCart 灰色图标 + 文字「购物车空空如也~」，底部结算栏隐藏，不显示0元 |
| 数量 = 1 时按「-」 | 商品数量是 1，点减号 | 不做任何变化，不会减到 0 |
